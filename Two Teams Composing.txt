#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
#define pb   push_back
void solve ()
{
    ll n;
    cin >> n;
    vector<ll > v(n);
    for (int i=0; i<n; i++)
    {
        cin >> v[i];
    }
     sort(v.begin() , v.end());
    ll lo, up;
    ll mx =0;
    for (int i=0;i<n; i++)
    {
        lo = lower_bound(v.begin() , v.end(), v[i])-v.begin();
        up = upper_bound(v.begin() , v.end(), v[i]) - v.begin();
        ll x = up - lo;
        mx = max(x, mx);
    }
      ll ans ;
    set<ll > ss;
    for (int i=0 ; i<n; i++)
    {
        ss.insert(v[i]);
    }
    ll y = ss.size();
    if (y==mx)
    {
        ans =  min(y, mx) ;
        ans--;
    }
    else {
          ans =  min(y, mx) ;
    }
    cout << ans << endl;

}
int main()
{
 ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int t;
    cin >> t;
    while (t--)
    {
        solve ();
    }
}
